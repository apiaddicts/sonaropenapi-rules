Se valida que cada ruta alterna sus partes entre partes estáticas y variables, empezando por las partes estáticas.

Ejemplos:

  • /{id} -> Error
  • /facturas/{id}/{lineId} -> Error
  • /facturas/{id} -> Ok
  • /facturas/get -> Ok
  • /facturas/{id}/lineas/{lineId} -> Ok
  • /facturas/{id}/lineas/get -> Ok

Ejemplo de código no compatible (OpenAPI 2)

JSON

{
    "swagger": "2.0", 
    "info": {
        "title": "Swagger Petstore", 
        "version": "1.0.0"
    }, 
    "paths": {
        "/customers/{customerId}/{invoiceId}": {
            "get": {
                "parameters": [
                    {
                        "in": "path", 
                        "name": "customerId", 
                        "type": "integer", 
                        "required": true
                    }, 
                    {
                        "in": "path", 
                        "name": "invoiceId", 
                        "type": "integer", 
                        "required": true
                    }
                ], 
                "responses": {
                    "200": {
                        "description": "Ok"
                    }
                }
            }
        }
    }
}

YAML

swagger: "2.0"
info:
  title: Swagger Petstore
  version: 1.0.0
paths:
  /customers/{customerId}/{invoiceId}:
    get:
      parameters:
        - in: path
          name: customerId
          type: integer
          required: true
        - in: path
          name: invoiceId
          type: integer
          required: true
      responses:
        200:
          description: Ok
