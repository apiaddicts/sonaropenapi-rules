Solo puede haber como máximo un parámetro body en cada operación. Esta restricción aplica únicamente a OpenAPI 2.

Ejemplo de código no compatible (OpenAPI 2)

JSON

{
    "swagger": "2.0",
    "info": {
        "version": "1.0.0",
        "title": "Swagger Petstore"
    },
    "paths": {
        "/pets": {
            "post": {
                "parameters": [
                    {
                        "in": "body",
                        "name": "someParam",
                        "required": true,
                        "schema": {
                            "type": "string"
                        }
                    },
                    {
                        "in": "body",
                        "name": "otherParam",
                        "required": true,
                        "schema": {
                            "type": "string"
                        }
                    }
                ],
                "responses": {
                    "default": {
                        "description": "the default response"
                    }
                }
            }
        }
    }
}

YAML

swagger: "2.0"
info:
  version: 1.0.0
  title: Swagger Petstore
paths:
  /pets:
    post:
      parameters:
      - in: body
        name: someParam
        required: true
        schema:
          type: string
      - in: body
        name: otherParam
        required: true
        schema:
          type: string
      responses:
        default:
          description: the default response

Solución compatible (OpenAPI 2)

JSON

{
    "swagger": "2.0",
    "info": {
        "version": "1.0.0",
        "title": "Swagger Petstore"
    },
    "paths": {
        "/pets": {
            "post": {
                "parameters": [
                    {
                        "in": "body",
                        "name": "someParam",
                        "required": true,
                        "schema": {
                            "type": "string"
                        }
                    },
                    {
                        "in": "query",
                        "name": "otherParam",
                        "required": true,
                        "schema": {
                            "type": "string"
                        }
                    }
                ],
                "responses": {
                    "default": {
                        "description": "the default response"
                    }
                }
            }
        }
    }
}

YAML

swagger: "2.0"
info:
  version: 1.0.0
  title: Swagger Petstore
paths:
  /pets:
    post:
      parameters:
      - in: body
        name: someParam
        required: true
        schema:
          type: string
      - in: query
        name: otherParam
        required: true
        schema:
          type: string
      responses:
        default:
          description: the default response
