The convention for naming properties in a JSON is "camelCase" (all words together and capitalizing the first letter of each word after the first), therefore, the path parameters, the query parameters, request body properties and response body properties must be in this convention.

Noncompliant Code Example (OpenAPI 2)

JSON

{
    "swagger": "2.0", 
    "info": {
        "title": "Swagger Petstore", 
        "version": "1.0"
    }, 
    "paths": {
        "/pets/{pet-id}": {
            "patch": {
                "consumes": [
                    "application/json"
                ], 
                "produces": [
                    "application/json"
                ], 
                "parameters": [
                    {
                        "in": "path", 
                        "name": "pet-id", 
                        "type": "integer", 
                        "required": true
                    }, 
                    {
                        "in": "query", 
                        "name": "pet_Type", 
                        "type": "string"
                    }, 
                    {
                        "in": "body", 
                        "name": "pet", 
                        "description": "Pet", 
                        "schema": {
                            "type": "object", 
                            "required": [
                                "PET_NAME"
                            ], 
                            "properties": {
                                "PET_NAME": {
                                    "type": "string"
                                }, 
                                "photoUrls": {
                                    "type": "array", 
                                    "items": {
                                        "type": "string"
                                    }
                                }
                            }
                        }
                    }
                ], 
                "responses": {
                    "200": {
                        "description": "OK", 
                        "schema": {
                            "type": "object", 
                            "properties": {
                                "petId": {
                                    "type": "integer"
                                }, 
                                "petType": {
                                    "type": "string"
                                }, 
                                "petName": {
                                    "type": "string"
                                }, 
                                "photoUrls": {
                                    "type": "array", 
                                    "items": {
                                        "type": "string"
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
    }
}
