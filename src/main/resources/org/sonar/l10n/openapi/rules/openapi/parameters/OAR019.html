Compliant Solution (OpenAPI 2)

  swagger: "2.0"
  info:
    version: 1.0.0
    title: Swagger Petstore
  paths:
    /pets:
      get:
        parameters:
          - in: query
            name: $select
            type: array
            items:
              type: string
          - in: query
            name: $select
            type: array
            items:
              type: string
